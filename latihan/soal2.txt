PROGRAM
    {Menentukan Kuadran suatu titik}
KAMUS
    x,y : integer

ALGORITME
    if(x = 0) then
        if(y = 0) then
            output("Kuadran tidak bisa dihitung")
        else
            output("Kuadran tidak bisa dihitung")
    else 
        if(y = 0) then
            if(x = 0) then
                output("Kuadran tidak bisa dihitung")
            else
                output("Kuadran tidak bisa dihitung")
        else
            if(x > 0 and y > 0) then
                output("Kuadran 1")
            else
                if(x < 0 and y > 0) then
                    output("Kuadran 2")
                else
                    if(x < 0 and y < 0) then
                        output("Kuadran 3")
                    else
                        output("Kuadran 4")
